// Переменные для хранения состояния калькулятора let currentInput = ''; let operation = ''; let firstOperand = null; // Функция для добавления значения в текущее поле ввода function appendToInput(value) { currentInput += value; updateInputField(); } // Функция для установки операции function setOperation(op) { if (currentInput === '') return; // Если первая операнда еще не установлена, устанавливаем ее if (firstOperand === null) { firstOperand = parseFloat(currentInput); } else { calculate(); } operation = op; currentInput = ''; } // Функция для выполнения вычислений function calculate() { // Проверяем, установлены ли операнды и операция if (firstOperand === null || currentInput === '') return; const secondOperand = parseFloat(currentInput); let result; // Выполняем вычисление в зависимости от операции switch (operation) { case '+': result = firstOperand + secondOperand; break; case '-': result = firstOperand - secondOperand; break; case '*': result = firstOperand * secondOperand; break; case '/': result = firstOperand / secondOperand; break; case 'sin': result = Math.sin(toRadians(secondOperand)); break; case 'cos': result = Math.cos(toRadians(secondOperand)); break; case 'tan': result = Math.tan(toRadians(secondOperand)); break; default: return; } // Обновляем текущее значение и сбрасываем операнды currentInput = result.toString(); firstOperand = null; operation = ''; updateInputField(); } // Функция для очистки поля ввода function clearInput() { currentInput = ''; firstOperand = null; operation = ''; updateInputField(); } // Функция для обновления поля ввода function updateInputField() { const inputs = document.querySelectorAll('input[type="text"]'); inputs.forEach(input => { input.value = currentInput; }); } // Функция для преобразования градусов в радианы function toRadians(degrees) { return degrees * (Math.PI / 180); } // Функция для отображения выбранного калькулятора function showCalculator(calcType) { const calculators = document.querySelectorAll('.calculator'); calculators.forEach(calc => { calc.classList.remove('active'); }); document.getElementById(calcType).classList.add('active'); const tabs = document.querySelectorAll('.tab-button'); tabs.forEach(tab => { tab.classList.remove('active'); }); document.querySelector(`.tab-button[onclick="showCalculator('${calcType}')"]`).classList.add('active'); }
